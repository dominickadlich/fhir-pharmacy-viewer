export interface ParsedMedication {
    id: string;
    name: string;
    status: string;
    authoredOn: string;
    sig: string;
    indication: string;
    refillsAllowed: number;
}

interface ParsedAllergy {
    id: string;
    code: string;
    criticality: string;
    reaction: string;
    recordedDate: string;
}

interface ParsedObservation {
    id: string;
    code: string;
    value: number | null;
    unit: string;
    referenceRange: string;
    effectiveDateTime: string;
}

export function parseObservation(entry: fhir4.Observation): ParsedObservation | null {
    if (!entry || entry.resourceType !== 'Observation') return null;

    return {
        id: entry.id ?? "",
        code: entry.code?.text ?? entry.code?.coding?.[0]?.display ?? "Unknown Observation",
        value: entry.valueQuantity?.value ?? null,
        unit: entry.valueQuantity?.unit ?? entry.valueQuantity?.code ?? "",
        referenceRange: entry.referenceRange?.[0]?.text ?? `${entry.referenceRange?.[0]?.low?.value ?? ""} - ${entry.referenceRange?.[0]?.high?.value ?? ""}`,
        effectiveDateTime: entry.effectiveDateTime ?? "",
    }
}

export function parseAllergy(entry: fhir4.AllergyIntolerance): ParsedAllergy | null {
    if (!entry || entry.resourceType !== "AllergyIntolerance") return null;

    return {
        id: entry.id ?? "",
        code: entry.code?.text ?? entry.code?.coding?.[0]?.display ?? "Unknown allergen",
        criticality: entry.criticality ?? "Unknown",
        reaction: entry.reaction?.[0]?.manifestation?.[0]?.text ?? "",
        recordedDate: entry.recordedDate ?? "",
    }
}

export function parseMedications(entry: fhir4.MedicationRequest): ParsedMedication | null {
    if (!entry || entry.resourceType !== "MedicationRequest") return null;
    
    const name = entry.medicationCodeableConcept?.text
        ?? entry.medicationCodeableConcept?.coding?.[0]?.display
        ?? (entry.medicationReference as fhir4.Reference)?.display
        ?? "Unknown medication";

    return {
        id: entry.id ?? "",
        name,
        status: entry.status ?? "",
        authoredOn: entry.authoredOn ?? "",
        sig: entry.dosageInstruction?.[0]?.text ?? "",
        indication: entry.reasonCode?.[0].text ?? "",
        refillsAllowed: entry.dispenseRequest?.numberOfRepeatsAllowed ?? 0
    }
}


app/lib/fhir/index.ts
import FHIR from "fhirclient";
import { parseAllergy, parseMedications, parseObservation } from "./parsers";
import Client from "fhirclient/lib/Client";

export default async function getPatientData() {
    const client: Client = await FHIR.oauth2.ready();

    const [patient, medBundle, allergyBundle, obsBundle] = await Promise.all([
        client.patient.read(),
        client.patient.request<fhir4.Bundle>("MedicationRequest"),
        client.patient.request<fhir4.Bundle>("AllergyIntolerance"),
        client.patient.request<fhir4.Bundle>("Observation?category=laboratory"),
    ]);

    return {
        patient,
        medication: medBundle.entry
            ?.map(e => parseMedications(e.resource as fhir4.MedicationRequest))
            .filter(Boolean) ?? [],
        allergies: allergyBundle.entry
            ?.map(e => parseAllergy(e.resource as fhir4.AllergyIntolerance))
            .filter(Boolean) ?? [],
        observations: obsBundle.entry
            ?.map(e => parseObservation(e.resource as fhir4.Observation))
            .filter(Boolean)
    }
}

## What broke / what confused me

**Interface design — passing the problem downstream**
My first pass at the interfaces tried to store raw FHIR codeable concept objects rather than resolved string values. That deferred the parsing work to whatever consumed the interface, which the wrong approach. The parser's job is to resolve raw FHIR structures into clean, typed primitives so the UI never has to touch optional chaining. The fix was straightforward: resolve each field to its display value inside the parser's return statement, with a sensible fallback at every level.

**`fhirclient` is a browser-only library**

After moving the data fetching logic into `getPatientData()`, I attempted to clean up the callback component by removing `'use client'`, `useEffect`, and `useRef`. This immediately threw:
```
TypeError: Cannot read properties of undefined (reading 'ready')
    at getPatientData (app/lib/fhir/index.ts:6:46)
    at CallbackPage (app/api/auth/callback/page.tsx:4:38)

The root cause: fhirclient depends on sessionStorage to manage the OAuth token, which is a browser API unavailable in Next.js server components. Moving the function to a separate file doesn't change its runtime context — Next.js will still attempt to execute it on the server unless the component that calls it is explicitly marked 'use client'. The 'use client' directive, useEffect, and useRef guard are all staying in the callback component for now.

## What I learned
Promise.all() is the correct pattern for parallel FHIR requests. Chaining .then() calls fires each request sequentially — each one waits for the previous to complete before starting. Promise.all() fires all four requests simultaneously and waits for the slowest one. For a patient snapshot that requires demographics, medications, allergies, and labs, sequential fetching is unnecessarily slow. Parallel fetching is both cleaner to read and meaningfully faster in practice.

The parser layer is what separates a FHIR integration from a FHIR-aware application. Raw FHIR Bundles are flexible by design — the spec accommodates enormous variation in how implementers populate fields. That flexibility is a liability in a UI. A ParsedMedication interface enforces a contract: the component that renders a medication card doesn't need to know whether the data came from medicationCodeableConcept or medicationReference. The parser absorbs that complexity so the UI doesn't have to.
